def measure_response_time(prompt):
    start_time = time.time()
    
    result = subprocess.run(
        ['claude', 'chat', prompt],
        capture_output=True,
        text=True
    )
    
    end_time = time.time()
    return {
        'prompt': prompt,
        'response_time': end_time - start_time,
        'output_length': len(result.stdout)
    }
